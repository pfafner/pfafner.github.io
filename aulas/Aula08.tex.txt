\section{Conjuntos Mesurables}

% ------------ Frame 01 ------------
\begin{frame}{Conjuntos Mesurables}

\small
\textbf{Ejemplo 5:} Todo cerrado $F \subseteq \RR^n$ es Lebesgue-mesurable. \\
\bigskip %\pause

\textbf{Prueba:} Sea $F \subseteq \RR^n$ un conjunto cerrado. %\pause
Separamos la prueba en dos casos:

\begin{itemize}
    \small
    \item Si $F$ es compacto, %\pause 
    dado $\varepsilon > 0$, tomemos $G \subseteq \RR^n$ un abierto tal que
    \vspace{-0.1cm}
    $$ F \subseteq G \qquad \textmd{y} \qquad |G| \leq |F|_e + \varepsilon. $$
    \vspace{-0.4cm} %\pause

    Como $G$ es abierto y $F$ cerrado, entonces $G-F$ es abierto. %\pause
    Entonces, $G-F$ puede escribirse como una unión enumerable disjunta de intervalos abiertos $G-F = \bigcup_k I_k$. %\pause
    En particular
    \vspace{-0.2cm}
    $$ |G-F|_e = |G-F| = \Big| \bigcup_k I_k \Big| \leq \sum_k |I_k|. $$
    \vspace{-0.2cm} %\pause
    
    Mostramos ahora que $\sum_k |I_k| < \varepsilon$. %\pause
    Observe que $F \cup \bigcup_k I_k \subseteq G$, %\pause
    y como $F$ y los $I_k$ son todos disjuntos, entonces
    $|F|_e + \sum_k |I_k|_e = \big| F \cup \bigcup_k I_k \big| \leq |G|$, %\pause
    De ahí que
    \vspace{-0.2cm}
    $$ \sum_k |I_k| \leq |G| - |F|_e < \varepsilon. $$
\end{itemize}
\end{frame}


% ------------ Frame 02 ------------
\begin{frame}{Conjuntos Mesurables}

\small
\begin{itemize}
    \small
    \item Para el caso general en que $F$ es un cerrado arbitrario, consideremos las intersecciones
    $$ F_k = F \cap \overline{\DD_k(0)} = \{\xx \in F: \ ||\xx|| \leq k \}, \qquad k = 1,2,3,\ldots $$
    \vspace{-0.2cm} %\pause

    Cada $F_k$ es cerrado, pues es la intersección de dos cerrados, 
    y es limitado. %\pause
    Luego, todos los $F_k$ son compactos, y del punto anterior, todos estos son Lebesgue-mesurables. %\pause
    Además, 
    $$ \bigcup_{k \geq 1} F_k %\pause
    = \bigcup_{k \geq 1} \big( F \cap \overline{\DD_k(0)} \big) %\pause
    = F \cap \Big( \bigcup_{k \geq 1} \overline{\DD_k(0)} \Big) %\pause
    = F \cap \RR^n = F. $$
    \vspace{-0.2cm} %\pause

    Así, $F$ es la unión enumerable de conjuntos mesurables. %\pause
    Portanto $F$ es mesurable. $\boxed{}$
\end{itemize}
\vspace{2cm}
\end{frame}


% ------------ Frame 03 ------------
\begin{frame}{Conjuntos Mesurables}

\small
\textbf{Ejemplo 6:} El complemento $E^c$ de cualquier conjunto $E \subseteq \RR^n$ Lebesgue-mesurable, es Lebesgue-mesurable. \\
\bigskip %\pause

\textbf{Prueba:} %\pause
Sea $E \subseteq \RR^n$ mesurable. %\pause
Para cada $k \in \ZZ^+$, considemos abiertos $G_k \subseteq \RR^n$ con
\vspace{-0.2cm}
$$ E \subseteq G_k \qquad \testmd{y} \qquad |G_k - E| < \tfrac{1}{k}. $$
\vspace{-0.4cm} %\pause

Tomamos $H = \displaystyle \bigcup_{k\geq 1} G_k^c$. %\pause
Entonces $H$ es mesurable, por ser unión enumerable de cerrados. \\ %\pause
Además,
\vspace{-0.1cm}
$\ \ \displaystyle E \subseteq G_k \ \forall k %\pause
\ \Rightarrow \ G_k^c \subseteq E^c \ \forall k %\pause
\ \Rightarrow \ H = \bigcup_{k\geq 1} G_k^c \subseteq E^c$. \\
\vspace{0.1cm} %\pause

Escribimos $E^c = H \cup (E^c - H) = H \cup Z$. %\pause
Entonces, para cada $k \geq 1$
\vspace{-0.2cm}
$$ Z = E^c - H %\pause
= E^c - \bigcup_k G_k^c %\pause
\subseteq E^c - G_k^c %\pause
= E^c \cap G_k %\pause
= G_k \cap E^c %\pause
= G_k - E, $$
\vspace{-0.4cm} %\pause

y $|Z|_e \leq |G_k-E| < \tfrac{1}{k}$, para todo $k\geq 1$. %\pause
Portanto $|Z|_e = 0$, y $Z$ es mesurable. %\pause
Así, $E^c = H \cup Z$ es unión de mesurables, y es mesurable. 
\end{frame}


% ------------ Frame 04 ------------
\begin{frame}{Conjuntos Mesurables}

\small
\textbf{Ejemplo 7:} Toda intersección enumerable $E = \bigcap_{k \geq 1} E_k$ de conjuntos Lebesgue-mesurables $E_k \subseteq \RR^n$, es Lebesgue-mesurable. \\
\bigskip %\pause

\textbf{Prueba:} %\pause
Sea $E = \displaystyle \bigcap_{k\geq 1} E_k$, donde los $E_k \subseteq \RR^n$ son mesurables, para $k=1,2,3,\ldots$. %\pause
De la propiedad anterior, los $E_k^c$ son todos mesurables, %\pause
y portanto la unión enumerable $\displaystyle \bigcup_k E_k^c$, es mesurable. %\pause

Luego, 
$$ E = \displaystyle \bigcap_{k\geq 1} E_k %\pause
= \Big( \bigcup_{k\geq 1} E_k^c \Big)^c $$
es el complemento de un mesurable. Portanto es Lebesgue-mesurable. $\boxed{}$
\vspace{2cm}
\end{frame}


% ------------ Frame 05 ------------
\begin{frame}{Conjuntos Mesurables}

\small
\textbf{Ejemplo 8:} Si $E_1, E_2 \subseteq \RR^n$ son Lebesgue-mesurables, entonces $E_1 - E_2$ es Lebesgue-mesurable. \\
\bigskip %\pause

\textbf{Preuba:} De la propiedad anterior, si $E_1, E_2$ son mesurables, entonces $E_2^c$ es también mesurable. %\pause
De ahí que $E_1 - E_2 = E_1 \cap E_2^c$ corresponde a una intersección de mesurables. %\pause
Portanto, es Lebesgue-mesurable. \\
\bigskip %\pause

Para mostrar la igualdad, observe que vale la unión disjunta
$E_1 = E_2 \cup (E_1 - E_2)$, %\pause
de modo que 
$$ |E_1|_e = |E_2|_e + |E_1 - E_2|_e. $$

Como todos estos conjuntos son mesurables, obtenemos
\vspace{-0.2cm}
$$ |E_1|= |E_1|_e = |E_2|_e + |E_1 - E_2|_e = |E_2| + |E_1 - E_2|, $$
de modo que $|E_1 - E_2| = |E_1| - |E_2|$. $\ \boxed{}$
\end{frame}


% ------------ Frame 06 ------------
\begin{frame}{Conjuntos Mesurables}

\small
\begin{defn}
Sea $X$ un conjunto no vacío. Una \textbf{$\sigma$-álgebra} $\mF$ en $X$, es una colección de subconjuntos de $X$ que satisface
\begin{itemize}
    \small
    \item[i)] $A \in \mF \ \Longrightarrow \ A^c \in \mF$,
    \item[ii)] $\{A_i\}_{i=1}^{\infty} \in \mF \ \Longrightarrow \ \bigcup_{i=1}^\infty A_i \in \mF$,
\end{itemize}
\end{defn}
\bigskip %\pause

\textbf{Nota:} La propiedad de que $\varnothing, X \in  \mF$, se deduce de las propiedades (i) y (ii) en la defición. %\pause
Basta tomar la colección vacía $L = \{\}$, y obtenemos que $\varnothing = \bigcup_{A_i \in L} A_i \in \mF$. %\pause
De (i), tenemos que $X = \varnothing^c \in \mF$. \\
\bigskip %\pause

\begin{pros}
La colección de conjuntos Lebesgue-mesurables de $\RR^n$ es una $\sigma$-álgebra. $\boxed{}$
\end{pros}
\end{frame}


% ------------ Frame 07 ------------
\begin{frame}{Conjuntos Mesurables}

\small
\begin{pros}
Sea $X$ un conjunto no vacío, y sea $\{\mF_\ell\}_\ell$ una colección arbitraria de $\sigma$-álgebras de $X$. Entonces, la intersección $\bigcap \mF_\ell$ es también una $\sigma$-álgebra.
\end{pros}


\end{frame}
