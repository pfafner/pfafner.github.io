\section{Medidas}


% ------------ Frame 01 ------------
\begin{frame}{Medidas Positivas}

\small
Deseamos generalizar las propiedades que posee la medida de Lebesgue. \\ 
\vspace{0.2cm} %\pause

\begin{defn}
Sea $X$ un conjunto no vacío, y $\mA$ una $\sigma$-álgebra en $X$. Una \textbf{medida (positiva)} en $X$ es una función $\mu: \mA \to [0, \infty)$ que satisface:
\begin{itemize}
	\small
	\item[i)]  $\mu(\varnothing) = 0$,
	\item[ii)] Para cualquier colección enumerable $\{A_k\}_{k\geq 1} \subseteq \mM$, de conjuntos disjuntos a pares ($A_i \cap A_j = \varnothing$, para $i \neq j$), vale
    $$ \mu \Big( \bigcup_{k \geq 1} A_k \Big) = \sum_{k\geq 1} \mu(A_k). \qquad \textmd{(} \sigma \textmd{-aditividad)} $$
\end{itemize}
\end{defn}
\vspace{-0.2cm} %\pause

\textbf{Obs!}
Cuando valen las condiciones (i) y (ii) anteriores, pero $\mA$ no es una $\sigma$-álgebra, decimos que $\mu$ es una \textbf{pre-medida}.
\end{frame}


% ------------ Frame 02 ------------
\begin{frame}{Medidas Positivas}

\small
\textbf{Obs!}
Siempre se requiere verificar que $\bigcup_k A_k \in \mA$. %\pause
Cuando $\mA$ es una $\sigma$-álgebra, y los $A_k \in \mA$ esto no es necesario, %\pause
pero en el caso de pre-medidas, se requiere más cuidado: \\ %\pause
\color{red}
\textbf{Antes de calcular $\mu(B)$, se debe verificar que $B \in \mA$.} \\
\bigskip %\pause

\color{black}
\begin{defn}
Sea $\mA$ una $\sigma$-álgebra en $X$. El par $(X, \mA)$ se llama un \textbf{espacio mesurable}. Cuando fijamos una medida $\mu: \mA \to \RR$, llamamos a la estructura $(X, \mA, \mu)$ un \textbf{espacio de medida}.
\end{defn}
\vspace{0.2cm} %\pause

\begin{defn}
Una \textbf{medida finita} (o \textbf{medida compacta}) es aquella donde $\mu(X) < \infty$. \\ 
\vspace{0.1cm} %\pause

Una \textbf{medida de probabilidad} es aquella donde $\mu(X) = 1$. En este caso, denotamos usualmente $\mu = \PP$, y al espacio $(X, \mA, \PP)$ le llamamos un \textbf{espacio de probabilidad}. \\
\vspace{0.1cm} %\pause

Una medida $\mu: \mA \to \RR$ es \textbf{$\sigma$-finita} si $\mA$ contiene alguna secuencia $\{A_k\}_{k\geq 1}$ tal que $A_k \nearrow X$ y $\mu(A_k) < \infty$ para todo $k \geq 1$.
\end{defn}
\end{frame}


% ------------ Frame 03 ------------
\begin{frame}{Medidas Positivas}

\small
\begin{thm}[Propiedades de medidas positivas]
Sea $(X, \mA, \mu)$ un espacio de medida, y sean $A, B, A_k, B_k \in \mA$, para todo $k \geq 1$. Entonces:
\begin{itemize}
    \small
    \item[1)] (aditividad) %\pause
    $A \cap B = \varnothing \ \Longrightarrow \ \mu(A \cup B) = \mu(A) + \mu(B)$. %\pause
    \item[2)] (monotonía) %\pause
    $A \subseteq B \ \Longrightarrow \ \mu(A) \leq \mu(B)$. %\pause
    \item[3)] (diferencia) %\pause
    $A \subseteq B$ y $\mu(B) < \infty \ \Longrightarrow \ \mu(B - A) = \mu(B) - \mu(A)$. %\pause
    \item[4)] (inclusión-exclusión) %\pause
    $\mu(A \cup B) = \mu(A) + \mu(B) - \mu (A \cap B)$. %\pause
    \item[5)] (sub-aditividad) %\pause
    Para todo $n \in \NN$, $\displaystyle \mu \Big( \bigcup_{k = 1}^n A_k \Big) \leq \sum_{k = 1}^n \mu(A_k)$. %\pause
    \item[6)] (continuidad inferior) %\pause
    $A_k \nearrow A \ \Longrightarrow \ \mu(A) = \lim_k \mu(A_k) = \sup_k \mu(A_k)$. %\pause
    \item[7)] (continuidad superior) %\pause
    $B_k \searrow B$ y $\mu(B) < \infty \ \Longrightarrow \ \mu(B) = \lim_k \mu(B_k) = \inf_k \mu(B_k)$. %\pause
    \item[8)] ($\sigma$-sub-aditividad) %\pause
    $\displaystyle \mu \Big( \bigcup_{k\geq 1} A_k \Big) \leq \sum_{k \geq 1} \mu(A_k)$.
\end{itemize}
\end{thm}

\end{frame}


% ------------ Frame 04 ------------
\begin{frame}{Medidas Positivas}

\small
\textbf{Prueba:} %\pause
(1) (aditividad) $A \cap B = \varnothing \ \Longrightarrow \ \mu(A \cup B) = \mu(A) + \mu(B)$. \\
\vspace{0.2cm} %\pause

Hacemos $A_1 = A$, $A_2 = B$ y $A_k = \varnothing$, para $k \geq 3$. %\pause
Entonces $\{A_k\}_k$ es una secuencia de conjuntos disjuntos a pares, cuya unión es $A \cup B$. %\pause
Del axioma (ii), entonces
$$ \mu(A \cup B) = \mu \Big( \bigcup_{k \geq 1} A_k \Big) = \sum_{k \geq 1} \mu(A_k) %\pause
= \mu(A) + \mu(B). $$
%\pause

(2) (monotonía) $A \subseteq B \ \Longrightarrow \ \mu(A) \leq \mu(B)$. \\
\vspace{0.2cm} %\pause

Como $A \subseteq B$, entonces $B = A \cup (B-A)$ es una unión disjunta. %\pause
Por la propiedad (1), y como $\mu$ es una medida positiva, tenemos
$$ \mu(B) = \mu(A) + \underbrace{\mu(B-A)}_{\geq 0} %\pause
\geq \mu(A). $$


(3) (diferencia) $A \subseteq B$ y $\mu(B) < \infty \ \Longrightarrow \ \mu(B - A) = \mu(B) - \mu(A)$. \\
\end{frame}


% ------------ Frame 05 ------------
\begin{frame}{Medidas Positivas}

\small
Como $A \subseteq B$ y $\mu(B) < \infty$, %\pause
de la propiedad de monotonìa (2), obtenemos que $\mu(A) < \infty$. %\pause
Restando $\mu(A)$ en ambos lados de la ecuación $\mu(B) = \mu(A) + \mu(B-A)$, %\pause
tenemos
$$ \mu(B) - \mu(A) = \mu(B-A). $$
%\pause

(4)] (inclusión-exclusión) $\mu(A \cup B) = \mu(A) + \mu(B) - \mu (A \cap B)$. \\
\vspace{0.2cm} %\pause

Para cualesquiera $A, B \in \mA$, podemos escribir la unión $A \cup B$ como unión disjunta de tres conjuntos en $\mA$:
$$ A \cup B = \big[A - (A \cap B) \big] \cup \big[B - (A \cap B) \big] \cup \big[A \cap B\big]. $$
%\pause

De nuevo, la propiedad (1) garantiza que
\begin{eqnarray*}
    \mu(A \cup B) &=& \mu \big(A - (A \cap B) \big) + \mu \big(B - (A \cap B) \big) + \mu \big(A \cap B\big) \\ %\pause
    &=& (\mu(A) - \mu(A \cap B)) +  (\mu(B) - \mu(A \cap B)) + \mu(A \cap B) \\ %\pause
    &=& \mu(A) + \mu(B) - \mu(A \cap B).
\end{eqnarray*}
\end{frame}


% ------------ Frame 06 ------------
\begin{frame}{Medidas Positivas}

\small
(5) (sub-aditividad) Para todo $n \in \NN$, $\displaystyle \mu \Big( \bigcup_{k = 1}^n A_k \Big) \leq \sum_{k = 1}^n \mu(A_k)$. \\
\vspace{0.2cm} %\pause

Probamos por inducción sobre $n$. %\pause 
Usando el principio de inclusión-exclusión (4):
$$ \mu(A_1 \cup A_2) = \mu(A_1) + \mu(A_2) - \underbrace{\mu(A_1 \cap A_2)}_{\geq 0} \leq \mu(A_1) + \mu(A_2). $$
%\pause

Asumiendo que $\displaystyle \mu(A_1 \cup \ldots \cup A_n) \leq \sum_{k=1}^n \mu(A_k)$, %\pause
de nuevo el principio de inclusión-exclusión %\pause
\begin{eqnarray*}
    \mu(A_1 \cup A_2 \cup \ldots \cup A_{n+1}) &=& \mu \Big( \bigcup_{k=1}^n A_k \Big) + \mu(A_{n+1}) - \mu \Big( \bigcup_{k=1}^n A_k  \cap A_{n+1} \Big) \\ %\pause
     &\leq & \mu \Big( \bigcup_{k=1}^n A_k \Big) + \mu(A_{n+1}) %\pause
    \ \leq \ \sum_{k=1}^n \mu(A_k) + \mu(A_{n+1}) %\pause
    \ = \ \sum_{k=1}^{n+1} \mu(A_k).
\end{eqnarray*}
\end{frame}


% ------------ Frame 07 ------------
\begin{frame}{Medidas Positivas}

\small
(6) (continuidad inferior) $A_k \nearrow A \ \Longrightarrow \ \mu(A) = \lim_k \mu(A_k) = \sup_k \mu(A_k)$. \\
\vspace{0.2cm} %\pause

Sea $A = \bigcup_{k\geq 1} A_k$. %\pause 
Consideremos los conjuntos, definidos por
\vspace{-0.4cm}
$$ B_1 = A_1, \ \ B_2 = A_2 - A_1, \ \ B_3 = A_3 - (A_1 \cup A_2), \ \ \ldots \ \ B_k = A_k - \bigcup_{i=1}^{k-1} A_i, \ \forall k \geq 2. $$
\vspace{-0.2cm} %\pause

Observe que todos los $B_k \in \mA$. %\pause
Además, por inducción es simple verificar que $\displaystyle \bigcup_{k=1}^n B_k = A_n$, $\forall n \in \NN$, %\pause
de modo que $\displaystyle \bigcupdot_{k \geq 1} B_k = \bigcup_{k \geq 1} A_k = A$. \\
\vspace{0.1cm} %\pause

Por $\sigma$-aditividad, tenemos
\vspace{-0.1cm}
\begin{eqnarray*}
    \mu(A) &=& \sum_{k\geq 1} \mu(B_k) %\pause
    \ = \lim_{n \to \infty} \sum_{k=1}^n \mu(B_k) %\pause
    \ = \lim_{n \to \infty} \mu \Big( \bigcup_{k=1}^n B_k \Big) %\pause
    = \lim_{n \to \infty} \mu(A_n).
\end{eqnarray*}
\end{frame}


% ------------ Frame 08 ------------
\begin{frame}{Medidas Positivas}

\small
(7) (continuidad superior) $B_k \searrow B$ y $\mu(B) < \infty \ \Longrightarrow \ \mu(B) = \lim_k \mu(B_k) = \inf_k \mu(B_k)$. \\
\vspace{0.2cm} %\pause

Como $B_k \searrow B$, entonces $B_k \subseteq B_1$, para todo $k \geq 1$. \\
\vspace{0.2cm} %\pause

En particular, de la propiedad de monotonía (2), $\mu(B_1) < \infty$ implica que $\mu(B_1 - B_k) < \infty$, para todo $k \geq 1$. \\ 
\vspace{0.2cm} %\pause

Además,
\vspace{-0.1cm}
$$ B_k \searrow B \ \Rightarrow \ B_1 - B_k \nearrow B_1 - B, $$
%\pause
y este último límite también tiene medida $\mu(B_1 - B) < \infty$. \\
\vspace{0.2cm} %\pause

Por (6) y la propiedad de diferencias (3), tenemos que
$$ \mu(B_1) - \lim_k \mu(B_k) = \lim_k \big( \mu(B_1) - \mu(B_k) \big) = \lim_k \mu(B_1 - B_k) = \mu(B_1 - B) = \mu(B_1) - \mu(B). $$
%\pause
Esto muestra que $\lim\limits_k \mu(B_k) = \mu(B)$. \\
\vspace{0.1cm} %\pause

\end{frame}


% ------------ Frame 09 ------------
\begin{frame}{Medidas Positivas}

\small
(8) ($\sigma$-sub-aditividad) $\displaystyle \mu \Big( \bigcup_{k\geq 1} A_k \Big) \leq \sum_{k \geq 1} \mu(A_k)$. \\
\vspace{0.2cm} %\pause

De la sub-aditividad (5), tenemos $\displaystyle \mu \Big( \bigcup_{k=1}^n A_k \Big) \leq \sum_{k=1}^n \mu(A_k)$. %\pause
Luego,
$$ \mu \Big( \bigcup_{k=1}^\infty A_k \Big) = \mu \Big( \lim_n \bigcup_{k=1}^n A_k \Big) = \lim_n \mu \Big( \bigcup_{k=1}^n A_k \Big) \leq \lim_n \sum_{k=1}^n \mu(A_k) \leq \sum_{k=1}^\infty \mu(A_k). \ \boxed{} $$
\vspace{2cm} 
\end{frame}
